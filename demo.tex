% demo.tex — test harness for the Clean theme.
% Compiles with pdflatex (default in CI). See README.md for the optional
% LuaLaTeX / Fira Code route.
\documentclass[aspectratio=169, 11pt]{beamer}

% --- Fonts: Roboto Light (text) + Fira Mono (code) -------------------
% These packages ship the fonts inside TeX Live, so no system-font install
% is needed and they work under pdflatex.
% fontenc must come first: roboto's [light] option switches the default series
% to l (light), and if fontenc loads afterwards it re-selects the current font
% while still in OT1/cmss, which has no light series — yielding a
% "Font shape `OT1/cmss/l/n' undefined" substitution warning.
\usepackage[T1]{fontenc}
\usepackage[sfdefault, light]{roboto}
\usepackage[scaled]{FiraMono}
\usepackage{microtype}

% --- Bibliography: true Chicago author–date via biblatex-chicago -----
% biblatex-chicago loads biblatex with the Chicago authordate style; biber
% is the backend and latexmk runs it automatically in CI.
% natbib=true additionally exposes the familiar \citet / \citep commands
% (mapped onto the Chicago author–date engine), alongside biblatex's own
% \textcite / \parencite.
\usepackage[authordate, backend=biber, natbib=true]{biblatex-chicago}
\addbibresource{references.bib}

\usetheme{Clean}

\title{A title}
\subtitle{A subtitle}
\date{March 16, 2025}

% Richer author grid: \cleanauthor{name}{orcid}{email}{affiliation}.
% orcid/email/affiliation are optional — leave the group empty to omit.
\cleanauthor{Your Name}{0009-0007-1967-8304}{you@your-institution.edu}{Your Institution}
\cleanauthor{Coauthor Name}{}{coauthor@other-institution.edu}{Coauthor Institution}

\begin{document}

  \begin{itemize}

    \item \textcolor{cleanPrimary}{custom-coloured text} via \texttt{\textbackslash textcolor}.
  \end{itemize}

\begin{frame}{Blocks}{Standard, alert, and example blocks all stay on-palette}
  \begin{block}{A standard block}

  \end{block}
  \begin{alertblock}{An alert block}
    Alert blocks use the secondary accent.
  \end{alertblock}
  \begin{exampleblock}{An example block}
    Example blocks stay within the theme palette too (not Beamer's default green).
  \end{exampleblock}

\begin{frame}{Citations}{Author--year, resolved from a \texttt{.bib} file}
  Citations come straight from \texttt{references.bib}. The natbib-style
  commands are available alongside biblatex's own:
  \begin{itemize}
    \item Textual: \citet{callaway2021} study staggered adoption.
    \item Parenthetical: a robustness check \citep{goodman-bacon2021}.
    \item Multiple, sorted: \citep{callaway2021,arkhangelsky2024}.
    \item biblatex equivalents also work: \textcite{arkhangelsky2024},
      \parencite{goodman-bacon2021}.
  \end{itemize}
\end{frame}
