% demo.tex — test harness for the Clean theme.
% Compiles with pdflatex (default in CI). See CLAUDE.md for the lualatex route.
\documentclass[aspectratio=169, 11pt]{beamer}

% --- Fonts: Roboto Light (text) + Fira Mono (code) -------------------
% These packages ship the fonts inside TeX Live, so no system-font install
% is needed and they work under pdflatex.
\usepackage[sfdefault, light]{roboto}
\usepackage[scaled]{FiraMono}
\usepackage[T1]{fontenc}
\usepackage{microtype}

% --- Bibliography: true Chicago author–date via biblatex-chicago -----
% biblatex-chicago loads biblatex with the Chicago authordate style; biber
% is the backend and latexmk runs it automatically in CI.
\usepackage[authordate, backend=biber]{biblatex-chicago}
\addbibresource{references.bib}

\usetheme{Clean}

\title{A title}
\subtitle{A subtitle}
\date{March 16, 2025}

% Richer author grid: \cleanauthor{name}{orcid}{email}{affiliation}.
% orcid/email/affiliation are optional — leave the group empty to omit.
\cleanauthor{Your Name}{0009-0007-1967-8304}{you@your-institution.edu}{Your Institution}
\cleanauthor{Coauthor Name}{}{coauthor@other-institution.edu}{Coauthor Institution}

\begin{document}

\begin{frame}{Quarto Clean Typst Theme}{LaTeX / Beamer port}
  This deck is a Beamer re-implementation of the Touying clean theme.
  \begin{itemize}
    \item Left-aligned titles, primary accent, triangle list markers.
    \item Slide counter \(x / y\) in the footer.
